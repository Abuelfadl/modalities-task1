FilePath="D:\uni\Senior 2 Term 1\Imaging modalities\Task 1\Dataset"; %file path where data set is present
[V,sp] = dicomreadVolume(FilePath);
V = squeeze(V);
V = padarray(V,[0 0 139],0,'both');

%Dicom Read Volume function: Reading Dicom Images and creates a 4-D Dicom volume
%Video = VideoWriter('Dicom Video.mp4', 'MPEG-4'); 
% Video Writer: Creates an object to write video files
%Video.FrameRate = 60; 
%sets the frame rate of the video to 60 FPS
%open(Video);

for f = 1:size(V, 3)  
    imshow(mat2gray(rot90(rot90(permute(V(:,:,f),[1 2 3]))))); 
%permute function decides which two axes to be shown, here we only need the
%x-y plane (Axial plane)
end


for f = 1:size(V, 1)  
    imshow(mat2gray(rot90(permute(V(f,:,:),[2 3 1]))));
%regarding the permute fucntion we here convert the y axis to x and the z
%axis to y, therefore we are showing the y-z plane (Sagittal plane)
end

for f = 1:size(V, 2)  
    imshow(mat2gray(rot90(permute(V(:,f,:),[1 3 2]))));
%regarding the permute fucntion we here convert the z axis to y
%therefore we are showing the x and z plane (coronal plane)
end
%close(Video);
